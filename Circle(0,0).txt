#include <stdio.h>
#include <GL/gl.h>
#include <GL/glut.h>
#include <math.h>

int a,b,r;
void inputData()
{
    printf("Enter Center point: ");
    scanf("%d %d",&a,&b);
    printf("Enter Radius: ");
    scanf("%d",&r);

}

void Circle(void)
{
glClear (GL_COLOR_BUFFER_BIT);
glColor3f (0.0, 0.0, 0.0);
glPointSize(3.0);
float d,x,y;
d = 1-r;
x=0;y=r;
while(y>x)
{
    if(d<0)
    {
        d = d + (2*x) + 3;
    }
    else
    {
        d = d + (2*(x-y)) + 5;
        y--;
    }
    glBegin(GL_POINTS);
    glVertex2i(x,y);
    glVertex2i(-x,y);
    glVertex2i(x,-y);
    glVertex2i(-x,-y);
    glVertex2i(y,x);
    glVertex2i(-y,x);
    glVertex2i(y,-x);
    glVertex2i(-y,-x);
    glEnd();
    x++;
}
glFlush ();
}

void myInit (void)
{
glClearColor(1.0, 1.0, 1.0, 0.0);
glColor3f(0.0f, 0.0f, 0.0f);
glPointSize(4.0);
glMatrixMode(GL_PROJECTION);
glLoadIdentity();
gluOrtho2D(-300.0, 300.0, -300.0, 300.0);
}

main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize (600, 600);
    glutInitWindowPosition (100, 100);
    glutCreateWindow ("Circle Algorithm's");
    inputData();
    glutDisplayFunc(Circle);
    myInit ();
    glutMainLoop();


}
